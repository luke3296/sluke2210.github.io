Merge Sort

function mergeSort(T[low...high])
    Array[] left, right full
    if(low == high)
        return T[low]
    mid := floor((low+high)/2)
    left := mergeSort(T[low...mid])
    right:= mergeSort(T[mid+1...high])
    full:=merge(left,right)
    return full;
